193  vector<Tache*> L;
198  vector<Tache*> Q;
199  for (auto & p : etats_initiaux)
200  Q.push_back(taches_map_t[p]);
201 
202  while (!Q.empty()) {
203  Tache *n = Q.back();
204  Q.pop_back();
205  L.push_back(n);
206  for (auto & m : n->succ) {
207  auto it = map_arc.find( { n->nom, m });
208  map_arc.erase(it);
209 
210  string &str = taches_map_t[m]->pred;
211  str.erase(remove(str.begin(), str.end(), n->nom), str.end());
212 
213  string &str2 = n->succ;
214  str2.erase(remove(str.begin(), str.end(), m), str.end());
215 
216  if (taches_map_t[m]->pred.empty())
217  Q.push_back(taches_map_t[m]);
218  }
219  }
220  if (!map_arc.empty()) {
221  cout << "Des circuits existent !" << endl;
222  return true;
223  } else {
224  cout << "\nOrdre apres tri topologique : ";
